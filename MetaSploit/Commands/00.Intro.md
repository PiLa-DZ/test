# Metasploit Commands

```bash
# -------------------------------------------------------------
msfconsole -h
# -q, --quiet                      Do not print the banner on startup
# Example
msfconsole -q
msf >

# -------------------------------------------------------------
msf > help
```

```bash
show all
show auxiliary
show exploits
show payloads
show encoders
show nops
show post
show plugins
show info
show options
show favorites

# Open
use "module name"
# Exit
msf exploit(module name) > back

# Examples

# -------------------------------------------------------------
msf > search ftp
# -------------------------------------------------------------
msf > search vsftpd
msf > use unix/ftp/vsftpd_234_backdoor
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > info
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > show advanced
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > show targets
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > show evasion
# Module evasion options:
#    Name                Current Setting  Required  Description
#    ----                ---------------  --------  -----------
#    TCP::max_send_size  0                no        Maxiumum tcp segment size.  (0 = disable)
#    TCP::send_delay     0                no        Delays inserted before every send.  (0 = disable)
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > show payloads
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > set RHOSTS 192.168.0.199
msf exploit(unix/ftp/vsftpd_234_backdoor) > get RHOSTS
# RHOSTS => 192.168.0.199
# -------------------------------------------------------------
msf exploit(unix/ftp/vsftpd_234_backdoor) > unset RHOSTS
# Unsetting RHOSTS...
msf exploit(unix/ftp/vsftpd_234_backdoor) > get RHOSTS
# RHOSTS =>
# -------------------------------------------------------------
# NOTE: Set Global for all `RHOSTS` in Metasploit
msf exploit(unix/ftp/vsftpd_234_backdoor) > setg RHOSTS 192.168.0.199
# RHOSTS => 192.168.0.199
# -------------------------------------------------------------
# NOTE: reset Global for all `RHOSTS` in Metasploit
msf exploit(unix/ftp/vsftpd_234_backdoor) > unsetg RHOSTS 192.168.0.199
# Unsetting RHOSTS...
# Unsetting 192.168.0.199...



# =============================================================

# -------------------------------------------------------------
# Show all sessions
msf exploit(unix/ftp/vsftpd_234_backdoor) > sessions
# Active sessions
# ===============
# No active sessions.

# -------------------------------------------------------------
# Show all sessions
msf exploit(unix/ftp/vsftpd_234_backdoor) > sessions -i
# Active sessions
# ===============
# No active sessions.

# -------------------------------------------------------------
# Open sessions ID 3 for Example
msf exploit(unix/ftp/vsftpd_234_backdoor) > sessions -i 3

```
